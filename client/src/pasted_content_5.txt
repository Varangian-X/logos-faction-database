 * 2. Columns: A:ID, B:Name, C:Doc_ID, D:Stats_JSON, E:Inventory_JSON, F:Location

 * 4. Add key to Script Properties as 'GEMINI_API_KEY'
 */

const GEMINI_API_KEY = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
const MODEL_ID = 'gemini-2.0-flash'; // Fast, cheap model

/**
 * CORE FUNCTION: Handles a player action submission.
 * Trigger this from a Google Form submission or a Web App GET request.
 */
function handlePlayerAction(playerId, actionText) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const playerSheet = ss.getSheetByName("Players");
  
  // 1. Fetch Player Data
  const data = playerSheet.getDataRange().getValues();
  // Simple lookup (in reality, use a better search)
  let playerRowIndex = -1;
  let playerData = null;
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] == playerId) {
      playerRowIndex = i + 1; // 1-based index
      playerData = {
        name: data[i][1],
        docId: data[i][2],
        stats: JSON.parse(data[i][3]),
        inventory: JSON.parse(data[i][4]),
        location: data[i][5]
      };
      break;
    }
  }
  
  if (!playerData) return "Error: Player not found.";

  // 2. Construct the Prompt for the AI GM
  const systemPrompt = `
    You are 'The Logos', the AI Game Master of New Byzantium.
    Current Location: ${playerData.location}
    Player Stats: ${JSON.stringify(playerData.stats)}
    Player Inventory: ${JSON.stringify(playerData.inventory)}
    
    The player attempts: "${actionText}"
    
    Using the logic of 'The Architecture of Order' and 'Imperial Strategic Compendium':
    1. Determine the outcome (Success/Failure) based on stats.
    2. Write a short, atmospheric narrative paragraph describing the result.
    3. Update any stats or inventory if changed.
    
    Return ONLY JSON format:
    {
      "narrative": "The text...",
      "new_stats": {},
      "new_inventory": {},
      "new_location": "Current or new location"
    }
  `;

  // 3. Call Gemini API
  const aiResponse = callGeminiAPI(systemPrompt);
  const result = JSON.parse(aiResponse); // Parse the AI's JSON decision

  // 4. Update Database (Sheet)
  if (result.new_stats) playerSheet.getRange(playerRowIndex, 4).setValue(JSON.stringify(result.new_stats));
  if (result.new_inventory) playerSheet.getRange(playerRowIndex, 5).setValue(JSON.stringify(result.new_inventory));
  if (result.new_location) playerSheet.getRange(playerRowIndex, 6).setValue(result.new_location);

  // 5. Update Interface (Doc)
  if (playerData.docId) {
    const doc = DocumentApp.openById(playerData.docId);
    const body = doc.getBody();
    
    // Formatting the entry
    const title = body.appendParagraph(`> ACTION: ${actionText}`);
    title.setHeading(DocumentApp.ParagraphHeading.HEADING_3);
    title.setForegroundColor("#fbbf24"); // Imperial Gold color

    const narrative = body.appendParagraph(result.narrative);
    narrative.setHeading(DocumentApp.ParagraphHeading.NORMAL);
  }

  return "Turn Processed.";
}

/**
 * Helper: Calls Gemini API
 */
function callGeminiAPI(prompt) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${MODEL_ID}:generateContent?key=${GEMINI_API_KEY}`;
  
  const payload = {
    "contents": [{
      "parts": [{"text": prompt}]
    }],
    "generationConfig": {
      "response_mime_type": "application/json"
    }
  };

  const options = {
    "method": "post",
    "contentType": "application/json",
    "payload": JSON.stringify(payload)
  };

  const response = UrlFetchApp.fetch(url, options);
  const json = JSON.parse(response.getContentText());
  return json.candidates[0].content.parts[0].text;
}

/**
 * AUTOMATION: The Faction Turn
 * Run this weekly via 'Triggers'.
 */
function runWeeklyFactionTurn() {
  const factions = ["Neo-Praetorians", "Agentes en Rebus", "Neo-Varangians"];
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const worldSheet = ss.getSheetByName("World_State");
  
  // Logic: Each faction makes a move
  // In a real build, we'd read their goals from the sheet
  
  const prompt = `
    Generate a 'Faction Turn' report for New Byzantium.
    Factions: ${factions.join(", ")}.
    
    For each faction, decide one covert or overt move they make this week based on 'Factions Guide.pdf'.
    Return a summary paragraph.
  `;
  
  // We use standard text generation here, not JSON
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${MODEL_ID}:generateContent?key=${GEMINI_API_KEY}`;
  const payload = { "contents": [{ "parts": [{"text": prompt}] }] };
  
  const response = UrlFetchApp.fetch(url, {
    "method": "post", 
    "contentType": "application/json", 
    "payload": JSON.stringify(payload)
  });
  
  const text = JSON.parse(response.getContentText()).candidates[0].content.parts[0].text;
  
  // Log to World State
  worldSheet.appendRow([new Date(), text]);
}
