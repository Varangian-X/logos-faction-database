    <script>
        // Mock function to simulate Apps Script 'google.script.run'
        // In production, this connects to the backend described in Section 4.
        function handleAction(e) {
            e.preventDefault();
            const input = document.getElementById('command-input');
            const log = document.getElementById('terminal-log');
            const cmd = input.value.trim();

            if (!cmd) return;

            // 1. Display User Command
            const userEntry = document.createElement('p');
            userEntry.className = "text-imperial-gold mt-2";
            userEntry.textContent = `> USER: ${cmd}`;
            log.appendChild(userEntry);

            // 2. Simulate Processing (Loading State)
            const processingEntry = document.createElement('p');
            processingEntry.className = "text-hologram-blue opacity-75 animate-pulse";
            processingEntry.textContent = "> PROCESSING [LOGOS_ENGINE]...";
            log.appendChild(processingEntry);
            
            // Scroll to bottom
            log.scrollTop = log.scrollHeight;
            input.value = "";

            // 3. Simulate Response (Gemini API / Apps Script return)
            setTimeout(() => {
                log.removeChild(processingEntry); // Remove loading text
                
                const responseEntry = document.createElement('p');
                responseEntry.className = "text-white";
                
                // Simple keyword detection for demo purposes
                if (cmd.toLowerCase().includes("travel")) {
                    responseEntry.innerHTML = `> TRAVEL PROTOCOL INITIATED.<br>> Ticket purchased (-50 Credits).<br>> Arriving at destination in 1 Cycle.`;
                } else if (cmd.toLowerCase().includes("indulgence")) {
                    responseEntry.innerHTML = `> TRANSACTION COMPLETE.<br>> Sin absolved. Mind Stress reduced by 10%.<br>> The Church thanks you for your tithe.`;
                } else {
                    responseEntry.innerHTML = `> COMMAND ACKNOWLEDGED.<br>> Narrative log updated in Main Archive.<br>> Awaiting further instructions.`;
                }
                
                log.appendChild(responseEntry);
                log.scrollTop = log.scrollHeight;
            }, 1500); // 1.5s simulated delay
        }

        // Glitch Effect Helper (Optional visual flair mentioned in Section 5.1)
        setInterval(() => {
            const title = document.querySelector('h1');
            if (Math.random() > 0.95) {
                title.style.textShadow = "2px 2px 0px #ff3333, -2px -2px 0px #00f3ff";
                setTimeout(() => title.style.textShadow = "", 100);
            }
        }, 2000);
    </script>
