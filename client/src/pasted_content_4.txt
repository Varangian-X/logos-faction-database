    <script>
        // Simple Minigame Logic
        
        function selectZone(zone) {
            const display = document.getElementById('zone-info');
            const data = {
                'Chrysopolis': "CONTROL: Neo-Praetorians. STATUS: Secure. RESOURCES: High.",
                'Mese': "CONTROL: Contested. STATUS: Active Trading. RESOURCES: Medium.",
                'Cisterns': "CONTROL: Anarchy. STATUS: Riot Protocols. RESOURCES: Low."
            };
            display.innerText = `>> ANALYZING ${zone.toUpperCase()}...\n${data[zone]}`;
            log(`Scanning sector: ${zone}`);
        }

        function sendCommand() {
            const input = document.getElementById('cmd-input');
            const cmd = input.value.trim();
            if (cmd) {
                log(`> ${cmd}`);
                setTimeout(() => {
                    log(`PROCESSING: "${cmd}"...`);
                    // In a real app, this would call google.script.run.handlePlayerAction(cmd)
                    log(`RESPONSE: Directive acknowledged. Resources deducted.`);
                }, 600);
                input.value = '';
            }
        }

        function log(text) {
            const console = document.getElementById('console-output');
            console.innerHTML += `<br>> ${text}`;
            console.scrollTop = console.scrollHeight;
        }
    </script>
